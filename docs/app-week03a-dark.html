w02_lab01 : Dart 程式啟動函數 main()

// Define a function.
void printInteger(int aNumber) {
  print('The number is ' + aNumber.toString() + '.'); 
}

// This is where the app starts executing.

  var number = 42; // Declare and initialize a variable.
  printInteger(number); // Call a function.
}

void main(){
    int a = 13;
    if(a%2==0){
        print('$a is even number.');
    } else{
        print('$a is odd number.');
    }
}

w02_lab03 : 使用 import 載入 Dart 語言內建或第三方輔助套件

import 'dart:math';

Random rnd = new Random();

int randInt(int min, int max) {
  return min + rnd.nextInt(max - min + 1);
}

void main() {
  for (int i = 0; i < 10; i++) {
    print(randInt(1, 6));
  }
}

w02_lab04 : 像 Python 一樣提供常用內建資料結構 List 和 Map

void main() {
  List l = [];
  Map m = {'Python':0, 'Java':0, 'C/C++':0};
  for (int i = 1; i <= 6; i++) {
    l.add(i);
  }
  print("l = $l");
  l[2] = 77;
  print("l = $l");
  print("m = $m");
  m['Java']+=1;
  m['Python'] = 33;
  m['C/C++'] += 66;
  print("m = $m");  
}
